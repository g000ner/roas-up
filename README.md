# Киселев Дмитрий
# Нербходимые настройки и установки:
 - в файле php.ini включить поддержку нужного драйвера для подключения к БД (для СУБД MySQL - pdo_mysql)
 - для создания объекта QueryBuilder в конструктор нужно передать массив содержащий поля:
    - "dbtype" - префикс нужной БД (прописаны константами в классе QueryBuilder),
    - "dbname" - имя БД,
    - "host" - хост,
    - "username" - имя пользователя,
    - "password" - пароль
